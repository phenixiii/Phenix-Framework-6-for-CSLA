create view PH_USERROLEINFO_V as
SELECT PH_USER_ROLE.UR_ID,
       PH_USER_ROLE.UR_US_ID,
       PH_USER_ROLE.UR_RL_ID,
       PH_USER.US_ID,
       PH_USER.US_NAME,
       PH_USER.US_USERNUMBER,
       PH_ROLE.RL_ID,
       PH_ROLE.RL_NAME,
       PH_DEPARTMENT.DP_NAME,
       PH_DEPARTMENT.DP_ID
  FROM PH_USER_ROLE
  JOIN PH_USER
    ON PH_USER.US_ID = PH_USER_ROLE.UR_US_ID
  JOIN PH_ROLE
    ON PH_ROLE.RL_ID = PH_USER_ROLE.UR_RL_ID
  LEFT JOIN PH_DEPARTMENT
    ON PH_DEPARTMENT.DP_ID = PH_USER.US_DP_ID
